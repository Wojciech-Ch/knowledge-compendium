• Konkatenacja (łączenie):
    let combinedString = string1 + string2;

  • Długość stringa:
    let lengthOfString = someString.length;

  • Porównywanie stringów:
    let areEqual = string1 === string2;

  • Wyszukiwanie podstringów:
    let indexOfSubstring = string.indexOf("podstring");

  • Wyciąganie podstringów:
    let substring = string.substring(1, 4);

  • Zamiana znaków lub podstringów:
    let replacedString = string.replace("stary", "nowy");

  • Podział stringa:
    let splitString = string.split(",");

  • Usuwanie białych znaków:
    let trimmedString = string.trim();

  • Zamiana wielkości liter:
    let upperCaseString = string.toUpperCase();

  • Formatowanie stringów:
    let formattedString = `Wartość: $value`;

  • Sprawdzanie zawartości:
    let containsSubstring = string.includes("substring");

  • Iteracja po znakach:
    for (let char of string) console.log(char);

  • Sprawdzanie początku/końca:
    let startsWith = string.startsWith("start");

  • Wyrażenia regularne:
    let matches = string.match(/wyrażenie/);

  • Kodowanie i dekodowanie:
    let encodedString = encodeURIComponent(string);

    let reversedString = string.split("").reverse().join("");

    let count = string.split("znak").length - 1;

    let newString = string.replaceAll("stary", "nowy");
